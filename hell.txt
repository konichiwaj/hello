#include <stdio.h>

int main() {
    int n, i, j, temp;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    int a[n];

    for(i = 0; i < n; i++) {
        printf("Enter element %d: ", i+1);
        scanf("%d", &a[i]);
    }

    int min = a[0], max = a[0];

    for(i = 0; i < n; i++) {
        if(a[i] < min) min = a[i];
        if(a[i] > max) max = a[i];
    }

    for(i = 0; i < n; i++) {
        for(j = i+1; j < n; j++) {
            if(a[i] > a[j]) {
                temp = a[i];
                a[i] = a[j];
                a[j] = temp;
            }
        }
    }

    printf("Min = %d\nMax = %d\n", min, max);

    printf("Ascending: ");
    for(i = 0; i < n; i++) printf("%d ", a[i]);

    printf("\nDescending: ");
    for(i = n-1; i >= 0; i--) printf("%d ", a[i]);

    return 0;
}

even odd prime
#include <stdio.h>

int main() {
    int n, i, flag = 0;

    printf("Enter a number: ");
    scanf("%d", &n);

    if(n % 2 == 0)
        printf("Even\n");
    else
        printf("Odd\n");

    if(n <= 1) {
        printf("Not Prime");
    } else {
        for(i = 2; i < n; i++) {
            if(n % i == 0) {
                flag = 1;
                break;
            }
        }

        if(flag == 0)
            printf("Prime");
        else
            printf("Not Prime");
    }

    return 0;
}

palindrome
#include <stdio.h>

int main() {
    int n, r, rev = 0, temp;

    printf("Enter a number: ");
    scanf("%d", &n);

    temp = n;

    while(n > 0) {
        r = n % 10;
        rev = rev * 10 + r;
        n = n / 10;
    }

    if(temp == rev)
        printf("Palindrome");
    else
        printf("Not Palindrome");

    return 0;
}

matrix_add_sub
#include <stdio.h>

int main() {
    int r1, c1, r2, c2, i, j;

    printf("Enter rows and columns of first matrix: ");
    scanf("%d %d", &r1, &c1);

    printf("Enter rows and columns of second matrix: ");
    scanf("%d %d", &r2, &c2);

    if(r1 != r2 || c1 != c2) {
        printf("Addition/Subtraction not possible");
        return 0;
    }

    int a[r1][c1], b[r2][c2], add[r1][c1], sub[r1][c1];

    printf("Enter first matrix:\n");
    for(i = 0; i < r1; i++)
        for(j = 0; j < c1; j++)
            scanf("%d", &a[i][j]);

    printf("Enter second matrix:\n");
    for(i = 0; i < r2; i++)
        for(j = 0; j < c2; j++)
            scanf("%d", &b[i][j]);

    for(i = 0; i < r1; i++) {
        for(j = 0; j < c1; j++) {
            add[i][j] = a[i][j] + b[i][j];
            sub[i][j] = a[i][j] - b[i][j];
        }
    }

    printf("\nAddition:\n");
    for(i = 0; i < r1; i++) {
        for(j = 0; j < c1; j++)
            printf("%d ", add[i][j]);
        printf("\n");
    }

    printf("\nSubtraction:\n");
    for(i = 0; i < r1; i++) {
        for(j = 0; j < c1; j++)
            printf("%d ", sub[i][j]);
        printf("\n");
    }

    return 0;
}

#include <stdio.h>

int main() {
    int r1, c1, r2, c2, i, j, k;

    printf("Enter rows and columns of first matrix: ");
    scanf("%d %d", &r1, &c1);

    printf("Enter rows and columns of second matrix: ");
    scanf("%d %d", &r2, &c2);

    if(c1 != r2) {
        printf("Multiplication not possible");
        return 0;
    }

    int a[r1][c1], b[r2][c2], c[r1][c2];

    printf("Enter first matrix:\n");
    for(i = 0; i < r1; i++)
        for(j = 0; j < c1; j++)
            scanf("%d", &a[i][j]);

    printf("Enter second matrix:\n");
    for(i = 0; i < r2; i++)
        for(j = 0; j < c2; j++)
            scanf("%d", &b[i][j]);

    // initialize result matrix
    for(i = 0; i < r1; i++)
        for(j = 0; j < c2; j++)
            c[i][j] = 0;

    // multiplication
    for(i = 0; i < r1; i++) {
        for(j = 0; j < c2; j++) {
            for(k = 0; k < c1; k++) {
                c[i][j] = c[i][j] + a[i][k] * b[k][j];
            }
        }
    }

    printf("Result matrix:\n");
    for(i = 0; i < r1; i++) {
        for(j = 0; j < c2; j++)
            printf("%d ", c[i][j]);
        printf("\n");
    }

    return 0;
}
