### Q3. Stack Interface (Palindrome Check)
Using an interface to define stack operations to reverse a word and check for palindromes.

interface Stack {
    void push(char c);
    char pop();
    boolean isEmpty();
}

class MyStack implements Stack {
    char[] data = new char[100];
    int top = -1;

    public void push(char c) { data[++top] = c; }
    public char pop() { return data[top--]; }
    public boolean isEmpty() { return top == -1; }
}

public class PalindromeCheck {
    public static void main(String[] args) {
        String word = "radar";
        MyStack s = new MyStack();
        
        for(char c : word.toCharArray()) s.push(c);
        
        String reversed = "";
        while(!s.isEmpty()) reversed += s.pop();
        
        if(word.equals(reversed)) System.out.println(word + " is a Palindrome.");
        else System.out.println(word + " is not a Palindrome.");
    }
}

        // Usage: java FileSearch <option> <string> <filename>

        String option = "", searchStr = "", fileName = "";

        if (args.length == 3) {
            option = args[0];
            searchStr = args[1];
            fileName = args[2];
        } else if (args.length == 2) {
            searchStr = args[0];
            fileName = args[1];

### Q1. Simple Arithmetic Calculator
This program implements a grid-based calculator. [cite_start]It handles basic math operations and includes "Backspace" and "Clear" functionality[cite: 15, 17, 28].

```java
import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class SimpleCalculator extends JFrame implements ActionListener {
    JTextField display = new JTextField();
    String operator = "";
    double firstNum = 0;
    boolean startNewNumber = true;

    public SimpleCalculator() {
        setTitle("Calculator"); [cite_start]// [cite: 16]
        setLayout(new BorderLayout());
        
        display.setHorizontalAlignment(JTextField.RIGHT);
        display.setEditable(false);
        add(display, BorderLayout.NORTH);

        JPanel panel = new JPanel(new GridLayout(5, 4, 5, 5));
        String[] buttons = {
            [cite_start]"Backspace", "Clear", "", "", // [cite: 17, 28]
            [cite_start]"7", "8", "9", "/", // [cite: 18, 19, 20]
            [cite_start]"4", "5", "6", "*", // [cite: 21, 22, 23]
            [cite_start]"1", "2", "3", "-", // [cite: 24, 25, 26]
            [cite_start]"0", ".", "=", "+"  // [cite: 27, 30]
        };

        for (String text : buttons) {
            JButton btn = new JButton(text);
            if (!text.equals("")) {
                btn.addActionListener(this);
                panel.add(btn);
            } else {
                panel.add(new JLabel(""));
            }
        }

        add(panel, BorderLayout.CENTER);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setSize(350, 400);
        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        String cmd = e.getActionCommand();

        if (cmd.matches("[0-9.]")) {
            if (startNewNumber) { display.setText(cmd); startNewNumber = false; }
            else { display.setText(display.getText() + cmd); }
        } else if (cmd.equals("Clear")) {
            display.setText("");
            firstNum = 0;
        } else if (cmd.equals("Backspace")) {
            String txt = display.getText();
            if (txt.length() > 0) display.setText(txt.substring(0, txt.length() - 1));
        } else if (cmd.equals("=")) {
            double secondNum = Double.parseDouble(display.getText());
            if (operator.equals("+")) display.setText(String.valueOf(firstNum + secondNum));
            if (operator.equals("-")) display.setText(String.valueOf(firstNum - secondNum));
            if (operator.equals("*")) display.setText(String.valueOf(firstNum * secondNum));
            if (operator.equals("/")) display.setText(String.valueOf(firstNum / secondNum));
            startNewNumber = true;
        } else {
            firstNum = Double.parseDouble(display.getText());
            operator = cmd;
            startNewNumber = true;
        }
    }

    public static void main(String[] args) { new SimpleCalculator(); }
}
```
